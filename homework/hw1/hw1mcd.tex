&& lg(n^{1000}) \\ 
&& (lg n)^n \\
&& n^{lg n} \\
&& n^{lg lg n} \\
&& (lg n)^{(lg n)} \\
&& lg^{1000}n \\
&& n^2 \\
&& n lg n \\
&& n \\
&& lg n \\
&& n^{(1/lgn)} \\
&& (1+0.001)^n \\
&& lg_{1000} n \\
&& 1
\end{eqnarray}
